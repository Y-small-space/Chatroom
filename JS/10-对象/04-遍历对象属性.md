# 4.遍历对象属性

**for...in** 语句用于对数组或者对象的属性进行循环操作。

    let obj = {
        name: 'pink老师',
        age: 18,
        sex: '男',
        fn: function{
            
        }
    }
    console.log(obj.name);
    console.log(obj.age);
    console.log(obj.sex);

    for (变量 in 对象){

    }

    for (let k in obj){
        console.log(k); // k 变量 输出 得到的是 属性名
        console.log(obj[k]); // obj[k] 得到是 属性值
    }
